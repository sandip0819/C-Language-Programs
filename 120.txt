#include <stdio.h>

void rotateArray(int arr[], int n, int k) {
    for (int i = 0; i < k; i++) {
        int temp = arr[0];
        for (int j = 0; j < n - 1; j++) {
            arr[j] = arr[j + 1];
        }
        arr[n - 1] = temp;
    }
}

int main() {
    int array[] = {1, 2, 3, 4, 5};
    int n = sizeof(array) / sizeof(array[0]);
    int k = 2;

    rotateArray(array, n, k);

    printf("Array after rotating %d times:\n", k);
    for (int i = 0; i < n; i++) {
        printf("%d ", array[i]);
    }

    return 0;
}
